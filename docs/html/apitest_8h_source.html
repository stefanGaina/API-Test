49 /** ***************************************************************************************************
50  * @brief This defines the main function and parses the arguments for an input file. Initializes
51  * API-Test and parses commands until "quit" or "q" is inputed by the user.
52  * @param title: The text printed at the beginning of the line when waiting for the commands to be
53  * inputed in terminal and not file.
54  * @param buffer_size: The maximum length of the inputed command (if exceeded it will be truncated).
55  * @param handler_function: This function is called everytime a new command has been successfully got.
56  * @param print_help_function: This function is called when "help" or "h" is inputted by
57  * the user.
58  * @return void
59  * @see APITEST_HANDLE_COMMAND
60  *****************************************************************************************************/
61 #define APITEST_MAIN(title, buffer_size, handler_function, print_help_function) \
62  int main(int argc, char* argv[]) \
63  { \
64  const gchar* input_file_name = NULL; \
65  apitest_Handler_t handler = {}; \
66  if (1 == argc) \
67  { \
68  APITEST_INTERNAL_TERMINAL_MODE_PRINT(); \
69  } \
70  else \
71  { \
72  input_file_name = argv[1]; \
73  (void)g_fprintf(stdout, "Running in file mode!\n"); \
74  if (3 < argc) \
75  { \
76  (void)g_fprintf(stdout, "Extra parameters will be ignored!\n"); \
77  } \
78  } \
79  if (FALSE == apitest_init(&handler, title, input_file_name, buffer_size)) \
80  { \
81  return EXIT_FAILURE; \
82  } \
83  while (TRUE) \
84  { \
85  apitest_get_command(&handler); \
86  if (0 >= command.argc) \
87  { \
88  continue; \
89  } \
90  if (0 == g_strcmp0("help", command.argv[0]) || 0 == g_strcmp0("h", command.argv[0])) \
91  { \
92  print_help_function(); \
93  continue; \
94  } \
95  if (0 == g_strcmp0("quit", command.argv[0]) || 0 == g_strcmp0("q", command.argv[0])) \
96  { \
97  break; \
98  } \
99  handler_function(); \
100  } \
101  apitest_deinit(&handler); \
102  return EXIT_SUCCESS; \
103  }

105 /** ***************************************************************************************************
106  * @brief Checks if the first argument matches the function name and if the count of parameters is also
107  * correct calls another function with the name of the previous with "_test" postfix added (e.g. for
108  * example() an example_test() needs to be defined).
109  * @param function_name: The name of the function the user wants to call (he will have to input its name
110  * first).
111  * @param parameters_count: How many parameters need to be inputted by the user for the function call to
112  * be made successfully.
113  * @return void
114  *****************************************************************************************************/
115 #define APITEST_HANDLE_COMMAND(function_name, parameters_count) \
116  do \
117  { \
118  if (0 != strcmp(#function_name, command.argv[0])) \
119  { \
120  break; \
121  } \
122  if (parameters_count != command.argc - 1) \
123  { \
124  (void)g_fprintf(stdout, "Invalid number of parameters! (required: %" PRIu8 ")\n", parameters_count); \
125  return; \
126  } \
127  function_name##_test(); \
128  return; \
129  } \
130  while (FALSE)
